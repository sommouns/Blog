# 简单的实现observable

总体非常类似vue2.x的watcher依赖收集的设计思路

function deepProxy(target, handler) {
  if (typeof target !== 'object') return target

  // recusive
  for (let key in target) {
    if (typeof target[key] === 'object') {
      deepProxy(target[key], handler)
    }
  }

  return new Proxy(target, handler())
}

function createObservable(target) {
  let handler = () => ({
    get(obj, key) {
      return Reflect.get(obj, key)
    },
    set(obj, key, val) {
      return Reflect.set(obj, key, val)
    }
  })

  return deepProxy(target, handler)
}

let globalFn = null
let counter = 0
class Reaction {
  constructor() {
    this.id = counter++
    this.store = {}
  }

  collect() {
    this.store[this.id] = this.store[this.id] || []
    this.store[this.id].push(globalFn)
  }

  run() {
    this.store[this.id].forEach(fn => {
      fn()
    })
  }

  static start(fn) {
    globalFn = fn
  }

  static end() {
    globalFn = null
  }
}

export default Reaction

import Reaction from "./reaction";
export default function autorun(fn) {
  Reaction.start(fn)
  fn()
  Reaction.end()
}
