# jsonp

function jsonp({url, cb, params}) {
	return new Promise((resolve, reject) => {
		let tag = document.creatElement('script')
    window[cb] = function (data) {
      resolve(data)
      document.body.removeChild(tag)
    }
    let params = []
    for (let key in params) {
      params.push(key+'='+params[key])
    }
    
    tag.src = url + '?' + params.join('&')
    document.body.appendChild(tag)
	})
}

# cors

let whiteList = ['localhost: 3000']

app.use(function(req, res) {
  if (whiteList.includes(req.headers.origin)) {
    res.setHeader('Access-Control-Allow-Origin', whiteList) // 设置哪个源访问
    res.setHeader('Access-Control-Allow-Headers', 'name') // 允许带什么头
    res.setHeader('Access-Control-Allow-Methods', 'PUT') // 允许什么方法访问我
    res.setHeader('Access-Control-Allow-Credentials', true) // 跨域带上cookie凭证
    res.setHeader('Access-Control-Max-Age', 6000) // OPTIONS的存活时间
    res.setHeader('Access-Contorl-Expose-Headers', 'name') // 允许前端获取头
    if (req.method === 'OPTIONS') {
     	res.end()
    }
  }
})

# postMessage

// a.html

<iframe src="localhost: 4000" onload="load()" id="frame"></iframe>

<script>
	function load() {
    let frame = window.getElementById('frame')
    frame.contentWindow.postMessage('hello im a.html', 'localhost: 4000')
    
    frame.onmessage = function(e) {
      console.log(e.data)
    }
  }
</script>


// b.html
<script>
	window.onmessage = function(e) {
    console.log(e.data)
    e.source.postMessage('hi im b.html', 'localhost: 3000')
  }
</script>
