# ast

  • 遍历我们的语法（js）=》语法树：esprima
  • 遍历语法树（先序深度优先）=〉更改树的内容：estraverse
  • 生成新的内容：escodegen

# Case1: 修改函数名ast为sommouns

const esprima = require('esprima')
let code = `function ast() {}`

let tree = esprima.parseScript(code)

const estraverse = require('estraverse')
estraverse.traverse(tree, {
  enter(node) {
    if (node.type === 'Identifer') {
      node.name = 'sommouns'
    }
  }
})

let escodegen = require('escodegen')
let r = escodegen.generate(tree)
console.log(r)

# Case1: 箭头函数

let code = `let sum = (a, b) => {return a + b}`
const t = require('babel-types')
const babel = require('babel-ciore')
let ArrowPlugin = {
  visitor: {
    ArrowFunctionExpression(path) {
      let node = path.node
      let params = node.params
      let body = node.body
      
      if (!t.isBlockStatement(body)) {
        // 不是代码块 a => a + 1
        let returnStatement = t.returnStatement(body)
        body = t.blockStatement([returnStatement])
      }
      let funcs = t.functionExpression(null, params, body, false, false)
      path.replaceWith(funcs)
    }
  }
}

const r = babel.transform(code, {
  plugins: {
    ArrowPlugin
  }
})
console.log(r)
